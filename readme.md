###code
```ruby
def count_fib x, pp, p 
	if x == 2
		p + pp
    else
	   count_fib x - 1, p, p + pp
	end
end
def fib x
    neg = x < 0 ? -1 : 1
	x = x * neg
	if x == 0 || x == 1
		neg * x
	else
		neg * (count_fib x, 0 ,1)
	end
end

(-10..10).each { |x| print "#{x}:#{(fib x)} " }
```
